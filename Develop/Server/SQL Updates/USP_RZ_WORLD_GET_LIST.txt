ALTER PROC [dbo].[USP_RZ_WORLD_GET_LIST]
AS
    SET NOCOUNT ON;
    
    SELECT WORLD_ID
         , NAME
         , IP
         , TYPE
         , CUR_PLAYER_COUNT
         , MAX_PLAYER_COUNT
         , SERVABLE
         , 0 AS EXPIRE
      FROM dbo.RZ_GT_WORLD
